Puoi anche copiare questo codice e incollarlo manualmente in un segnalibro se preferisci:

javascript:(function(){
    const url = encodeURIComponent(window.location.href);
    const title = encodeURIComponent(document.title || '');
    const links = document.querySelectorAll('link[rel~=\'icon\']');
    const favicon = links.length ? encodeURIComponent(links[0].href) : '';
    const target = `https://lumo250.github.io/LinkZen/#add?addurl=${url}&addtitle=${title}&addicon=${favicon}`;
    window.open(target, '_blank');
  })()
